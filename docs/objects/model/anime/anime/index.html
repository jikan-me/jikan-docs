    Anime Model

    Usage¶

    <?php

$jikan = new Jikan\Jikan;

// Fetch One Piece's anime data
// Request: https://myanimelist.net/anime/21
$anime = $jikan->Anime(
    (new \Jikan\Request\Anime\AnimeRequest(21))
);

echo $anime->getTitle();
